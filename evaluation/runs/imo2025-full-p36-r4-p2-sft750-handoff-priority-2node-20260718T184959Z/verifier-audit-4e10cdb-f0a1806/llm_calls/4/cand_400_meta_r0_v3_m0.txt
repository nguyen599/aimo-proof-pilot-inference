\[
d_{k-3}=\frac{N}{d_4},\quad d_{k-2}=\frac{N}{d_3},\quad d_{k-1}=\frac{N}{d_2}.
\]

\[
f(N)=\frac{N}{d_2}+\frac{N}{d_3}+\frac{N}{d_4}.
